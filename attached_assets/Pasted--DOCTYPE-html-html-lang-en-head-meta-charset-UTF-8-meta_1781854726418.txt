<script>
  let currentPair = 'EURUSD';
  let currentPipSize = 0.0001;
  let currentContractSize = 100000;

  const DEFAULT_PIP_VALUES = {
    EURUSD: 10,
    GBPUSD: 10,
    USDJPY: 9.09,
    XAUUSD: 10,
    BTCUSD: 1,
    NAS100: 1,
  };

  function setPair(el, pair, pip, contract) {
    document.querySelectorAll('.pair-tab').forEach(t => t.classList.remove('active'));
    el.classList.add('active');
    currentPair = pair;
    currentPipSize = pip;
    currentContractSize = contract;
    const cf = document.getElementById('customFields');
    cf.style.display = pair === 'Custom' ? 'flex' : 'none';
  }

  function setRisk(val) {
    document.getElementById('riskPct').value = val;
    document.querySelectorAll('.risk-chip').forEach(c => c.classList.remove('active'));
    event.target.classList.add('active');
  }

  function getUSDNGN() {
    return parseFloat(document.getElementById('usdRate').value) || 1600;
  }

  function calculate() {
    let pair = currentPair;
    let pipSize = currentPipSize;
    let contractSize = currentContractSize;

    if (pair === 'Custom') {
      pair = document.getElementById('customPair').value || 'Custom';
      pipSize = parseFloat(document.getElementById('customPipSize').value);
      contractSize = parseFloat(document.getElementById('customContractSize').value);
      if (!pipSize || !contractSize) {
        alert('Please fill in custom pip size and contract size.');
        return;
      }
    }

    const balUSD = parseFloat(document.getElementById('balanceUSD').value) || 0;
    const balNGN = parseFloat(document.getElementById('balanceNGN').value) || 0;
    const rate = getUSDNGN();
    const riskPct = parseFloat(document.getElementById('riskPct').value) || 1;
    const slPips = parseFloat(document.getElementById('slPips').value);
    const tpPips = parseFloat(document.getElementById('tpPips').value) || 0;
    const direction = document.getElementById('direction').value;

    if (!slPips || slPips <= 0) {
      alert('Please enter a valid Stop Loss in pips.');
      return;
    }

    // Determine effective balance in USD
    let effectiveBalUSD = balUSD;
    if (!effectiveBalUSD && balNGN) {
      effectiveBalUSD = balNGN / rate;
    }

    if (!effectiveBalUSD) {
      alert('Please enter your account balance (USD or Naira).');
      return;
    }

    // Effective Naira balance
    let effectiveBalNGN = balNGN || effectiveBalUSD * rate;

    // Pip value per standard lot in USD
    let pipValuePerLot = parseFloat(document.getElementById('pipValueOverride').value);
    if (!pipValuePerLot) {
      pipValuePerLot = DEFAULT_PIP_VALUES[pair] || (pipSize * contractSize);
    }

    const riskUSD = (riskPct / 100) * effectiveBalUSD;
    const lotSize = riskUSD / (slPips * pipValuePerLot);
    const profitUSD = tpPips ? tpPips * pipValuePerLot * lotSize : null;
    const riskNGN = riskUSD * rate;
    const profitNGN = profitUSD ? profitUSD * rate : null;
    const rr = tpPips ? (tpPips / slPips).toFixed(2) : null;

    // Update result
    document.getElementById('resultTitle').textContent =
      `📊 ${pair} — ${direction.toUpperCase()}`;

    document.getElementById('rLot').textContent =
      lotSize < 0.01 ? lotSize.toFixed(4) + ' lots' : lotSize.toFixed(2) + ' lots';

    document.getElementById('rRiskUSD').textContent =
      `$${riskUSD.toFixed(2)} (${riskPct}%)`;

    document.getElementById('rRiskNGN').textContent =
      `₦${numberWithCommas(riskNGN.toFixed(0))}`;

    document.getElementById('rProfitUSD').textContent =
      profitUSD !== null ? `$${profitUSD.toFixed(2)}` : '— (no TP set)';

    document.getElementById('rProfitNGN').textContent =
      profitNGN !== null ? `₦${numberWithCommas(profitNGN.toFixed(0))}` : '—';

    document.getElementById('rRR').textContent =
      rr ? `1 : ${rr}` : '— (no TP set)';

    document.getElementById('rPipVal').textContent =
      `$${pipValuePerLot.toFixed(2)}/pip/lot`;

    document.getElementById('rSLTP').textContent =
      `${slPips} pips / ${tpPips || '—'} pips`;

    document.getElementById('rBal').textContent =
      `$${effectiveBalUSD.toFixed(2)} / ₦${numberWithCommas(effectiveBalNGN.toFixed(0))}`;

    // Risk bar
    const fill = document.getElementById('riskFill');
    const pct = Math.min(riskPct, 10);
    fill.style.width = (pct / 10 * 100) + '%';
    fill.style.background = riskPct <= 1 ? 'var(--tg-green)'
      : riskPct <= 2 ? 'var(--tg-yellow)'
      : 'var(--tg-red)';

    const bubble = document.getElementById('resultBubble');
    bubble.classList.add('visible');
    bubble.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }

  function resetForm() {
    document.getElementById('balanceUSD').value = '';
    document.getElementById('balanceNGN').value = '';
    document.getElementById('riskPct').value = 1;
    document.getElementById('slPips').value = '';
    document.getElementById('tpPips').value = '';
    document.getElementById('pipValueOverride').value = '';
    document.getElementById('resultBubble').classList.remove('visible');
    document.querySelectorAll('.risk-chip').forEach(c => c.classList.remove('active'));
    document.querySelector('.risk-chip:nth-child(2)').classList.add('active');
  }

  function copyResult() {
    const pair = document.getElementById('resultTitle').textContent;
    const lot = document.getElementById('rLot').textContent;
    const risk = document.getElementById('rRiskUSD').textContent;
    const riskN = document.getElementById('rRiskNGN').textContent;
    const profit = document.getElementById('rProfitUSD').textContent;
    const profitN = document.getElementById('rProfitNGN').textContent;
    const rr = document.getElementById('rRR').textContent;
    const sltp = document.getElementById('rSLTP').textContent;
    const bal = document.getElementById('rBal').textContent;

    const text = `${pair}
Lot Size: ${lot}
Risk: ${risk} | ${riskN}
Profit: ${profit} | ${profitN}
